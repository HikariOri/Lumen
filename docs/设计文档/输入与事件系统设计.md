一、先统一几个关键事实（非常重要）
=================

### 1️⃣ SDL3 是“事件队列驱动”的

*   SDL3 核心是一个**全局事件队列**
*   你必须在主线程：
    *   `SDL_PumpEvents`
    *   `SDL_PollEvent` / `WaitEvent` [DeepWiki](https://deepwiki.com/edwardgushchin/SDL3-CS/3.3-input-and-events?utm_source=chatgpt.com)

👉 结论：

> **输入是“事件流”，不是状态轮询系统**

* * *

### 2️⃣ ImGui 不是输入源，它是“输入消费者”

正确流程是：

```
while (SDL_PollEvent(&event))
{
    ImGui_ImplSDL3_ProcessEvent(&event);
    // 你的系统处理
}
```

👉 官方推荐：必须把 SDL\_Event 喂给 ImGui [GitCode](https://blog.gitcode.com/904216acf0a122e2e46baae82ac91bee.html?utm_source=chatgpt.com)

* * *

### 3️⃣ ImGui 需要“输入抢占权”

```
ImGui::GetIO().WantCaptureMouse
ImGui::GetIO().WantCaptureKeyboard
```

👉 含义：

*   true → UI 在用输入（不要给游戏）
*   false → 游戏可以用

* * *

### 4️⃣ 输入系统必须在“主线程”

否则：

*   ImGui event queue 会崩（真实坑） [GitHub](https://github.com/ocornut/imgui/issues/6895?utm_source=chatgpt.com)

```
namespace Engine
{

enum class Key : uint16_t
{
    Unknown = 0,

    W, A, S, D,
    Space,
    Escape,

    MouseLeft,
    MouseRight,
    MouseMiddle,

    COUNT

enum class InputEventType
{
    None,

    KeyDown,
    KeyUp,

    MouseButtonDown,
    MouseButtonUp,

    MouseMove,
    MouseWheel,

    WindowResize,
    Quit

struct KeyEvent
{
    Key key;
    bool repeat;

struct MouseMoveEvent
{
    float x, y;
    float dx, dy;

struct MouseButtonEvent
{
    Key button;
};

struct MouseWheelEvent
{
    float delta;
};

struct WindowResizeEvent
{
    uint32_t width;
    uint32_t height;
};

struct InputEvent
{
    InputEventType type;

    union
    {
        KeyEvent key;

        MouseWheelEvent mouseWheel;
```

四、InputState.h（状态系统）
====================

```
#pragma once

#include "InputTypes.h"
#include <array>

namespace Engine
{

struct InputState
{
    std::array<bool, (size_t)Key::COUNT> current{};
    std::array<bool, (size_t)Key::COUNT> previous{};

    float mouseX = 0;
    float mouseY = 0;

    float mouseDX = 0;
    float mouseDY = 0;

    float wheel = 0;

    void BeginFrame()
    {
        previous = current;
        mouseDX = 0;
        mouseDY = 0;
        wheel = 0;
    }

    bool IsDown(Key k) const
    {
        return current[(size_t)k];
    }

    bool IsPressed(Key k) const
    {
        return current[(size_t)k] && !previous[(size_t)k];
    }

    bool IsReleased(Key k) const
    {
        return !current[(size_t)k] && previous[(size_t)k];
    }
};

}
```

七、InputSystem.cpp（核心逻辑）
=======================

```
#include "InputSystem.h"
#include <imgui.h>

namespace Engine
{

void InputSystem::BeginFrame()
{
    m_State.BeginFrame();
}

Key InputSystem::TranslateKey(int key)
{
    switch (key)
    {
        case SDLK_w: return Key::W;
        case SDLK_a: return Key::A;
        case SDLK_s: return Key::S;
        case SDLK_d: return Key::D;
        case SDLK_SPACE: return Key::Space;
        case SDLK_ESCAPE: return Key::Escape;
    }
    return Key::Unknown;
}

void InputSystem::ProcessSDLEvent(const SDL_Event& e)
{
    ImGuiIO& io = ImGui::GetIO();

    switch (e.type)
    {
        case SDL_EVENT_QUIT:
        {
            // 可扩展为事件系统
            break;
        }

        case SDL_EVENT_KEY_DOWN:
        {
            Key key = TranslateKey(e.key.keysym.sym);

            if (key != Key::Unknown)
            {
                // ImGui抢占
                if (!io.WantCaptureKeyboard)
                {
                    m_State.current[(size_t)key] = true;
                }
            }
            break;
        }

        case SDL_EVENT_KEY_UP:
        {
            Key key = TranslateKey(e.key.keysym.sym);

            if (key != Key::Unknown)
            {
                if (!io.WantCaptureKeyboard)
                {
                    m_State.current[(size_t)key] = false;
                }
            }
            break;
        }

        case SDL_EVENT_MOUSE_BUTTON_DOWN:
        {
            if (!io.WantCaptureMouse)
            {
                if (e.button.button == SDL_BUTTON_LEFT)
                    m_State.current[(size_t)Key::MouseLeft] = true;
            }
            break;
        }

        case SDL_EVENT_MOUSE_BUTTON_UP:
        {
            if (!io.WantCaptureMouse)
            {
                if (e.button.button == SDL_BUTTON_LEFT)
                    m_State.current[(size_t)Key::MouseLeft] = false;
            }
            break;
        }

        case SDL_EVENT_MOUSE_MOTION:
        {
            if (!io.WantCaptureMouse)
            {
                m_State.mouseDX += e.motion.xrel;
                m_State.mouseDY += e.motion.yrel;

                m_State.mouseX = (float)e.motion.x;
                m_State.mouseY = (float)e.motion.y;
            }
            break;
        }

        case SDL_EVENT_MOUSE_WHEEL:
        {
            if (!io.WantCaptureMouse)
            {
                m_State.wheel += e.wheel.y;
            }
            break;
        }

        default:
            break;
    }
}

}
```

```
class ActionMap
{
public:
    void Bind(Action action, Key key)
    {
        m_Map[action].push_back({ key });
    }

    bool IsActive(Action action, const InputState& state) const
    {
        auto it = m_Map.find(action);
        if (it == m_Map.end()) return false;

        for (auto& b : it->second)
        {
            if (state.IsDown(b.key))
                return true;
        }
        return false;
    }

    bool IsPressed(Action action, const InputState& state) const
    {
        auto it = m_Map.find(action);
        if (it == m_Map.end()) return false;

        for (auto& b : it->second)
        {
            if (state.IsPressed(b.key))
                return true;
        }
        return false;
    }

private:
    std::unordered_map<Action, std::vector<Binding>> m_Map;
};
```

Context Stack（关键）

```
class InputContextStack
{
public:
    void Push(InputContext* ctx)
    {
        m_Stack.push_back(ctx);
    }

    void Pop()
    {
        m_Stack.pop_back();
    }

    bool IsActionActive(Action action, const InputState& state)
    {
        for (auto it = m_Stack.rbegin(); it != m_Stack.rend(); ++it)
        {
            auto* ctx = *it;

            if (ctx->actions.IsActive(action, state))
                return true;

            if (ctx->blockInput)
                break;
        }
        return false;
    }

private:
    std::vector<InputContext*> m_Stack;
};
```

```
Ray ScreenToRay(float mouseX, float mouseY, Camera& cam)
{
    // NDC
    float x = (2.0f * mouseX) / width - 1.0f;
    float y = 1.0f - (2.0f * mouseY) / height;

    glm::vec4 clip = { x, y, -1.0f, 1.0f };

    glm::vec4 view = inverse(cam.proj) * clip;
    view.z = -1.0f;
    view.w = 0.0f;

    glm::vec3 worldDir = normalize((inverse(cam.view) * view).xyz);

    return { cam.position, worldDir };
}
```

```
struct PickResult
{
    Entity entity;
    float distance;
};

class PickingSystem
{
public:
    std::optional<PickResult> Pick(Ray ray)
    {
        PickResult best{};
        best.distance = FLT_MAX;

        for (auto e : sceneEntities)
        {
            if (!IntersectAABB(ray, e.bounds))
                continue;

            float d;
            if (IntersectMesh(ray, e.mesh, d))
            {
                if (d < best.distance)
                {
                    best.entity = e;
                    best.distance = d;
                }
            }
        }

        if (best.distance < FLT_MAX)
            return best;

        return {};
    }
};
```

```
        RequestPick(mouseX, mouseY);
    }

    // 4. 读取 picking 结果（延迟一帧）

    }

    // 5. Gizmo
    UpdateGizmo();
}
```
